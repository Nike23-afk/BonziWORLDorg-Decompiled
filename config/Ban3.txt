const express = require('express');
const app = express();
const port = 3000;

// Hardcoded list of banned IPs
const bannedIPs = ['24.77.90.227'];

// Middleware to check if the user's IP is banned
app.use((req, res, next) => {
    const userIP = req.ip; // Get the IP address of the user

    if (bannedIPs.includes(userIP)) {
        return res.status(403).send('You are banned from this website.');
    }

    next(); // User is not banned; proceed to the next middleware/route
});

// Example home route
app.get('/', (req, res) => {
    res.send('Welcome to the site!');
});

// Start the server
app.listen(port, () => {
    console.log(`Server is running on http://localhost:${port}`);
});
